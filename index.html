class MinMax(): 
   def __init__(self, field1, field2): 
      self.min = field1 
      self.max = field2 
   def __str__(self): 
      return ”Min {min} - Max {max}”.format(min=str(self.min), max=str(self.max)) 
 
def minmax(data):
